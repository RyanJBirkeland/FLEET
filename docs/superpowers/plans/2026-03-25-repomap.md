# repomap Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `repomap`, a standalone CLI + browser tool that visualizes repository architecture as an interactive, expandable graph using AST analysis + LLM enrichment.

**Architecture:** Three-layer system (Parser → Enrichment → Visualization) following clean architecture with dependency injection. Parser uses TypeScript compiler API. Enrichment uses `@anthropic-ai/claude-agent-sdk` via factory pattern (modeled after claude-task-runner). Visualization is React + D3-force served by Express.

**Tech Stack:** TypeScript, React, D3-force, Express, WebSocket, tsup (CLI bundler), vitest, Playwright, `@anthropic-ai/claude-agent-sdk`

**Spec:** `docs/superpowers/specs/2026-03-25-repomap-design.md` (in BDE repo)

**Repo:** New standalone repo at `~/projects/repomap`

  target: string

  name: string
  rootPath: string
  analyzedAt: string
  languages: string[]
  nodeCount: number
  edgeCount: number
}

export interface RepoGraph {
  nodes: GraphNode[]
  edges: GraphEdge[]
  metadata: RepoMetadata
}

export interface GraphDelta {
  nodesAdded: GraphNode[]
  nodesRemoved: string[]
  nodesUpdated: GraphNode[]
  edgesAdded: GraphEdge[]
  edgesRemoved: Array<{ source: string; target: string }>
  edgesUpdated: GraphEdge[]

// src/parser/ignore.ts
import { readFileSync, existsSync } from 'fs'
import { join } from 'path'
import { minimatch } from 'minimatch'

// Note: add minimatch to dependencies

export interface IgnorePatterns {
  patterns: string[]
}

const ALWAYS_IGNORE = ['node_modules', '.git', '.repomap']

export function loadIgnorePatterns(rootPath: string): IgnorePatterns {
  const patterns: string[] = [...ALWAYS_IGNORE]

  const gitignorePath = join(rootPath, '.gitignore')
  if (existsSync(gitignorePath)) {
    const lines = readFileSync(gitignorePath, 'utf-8')
      .split('\n')
      .map((l) => l.trim())
      .filter((l) => l && !l.startsWith('#'))
    patterns.push(...lines)
  }

  const repomapignorePath = join(rootPath, '.repomapignore')
  if (existsSync(repomapignorePath)) {
    const lines = readFileSync(repomapignorePath, 'utf-8')
      .split('\n')
      .map((l) => l.trim())
      .filter((l) => l && !l.startsWith('#'))
    patterns.push(...lines)
  }

  return { patterns }
}

export function shouldIgnore(relativePath: string, { patterns }: IgnorePatterns): boolean {
  for (const pattern of patterns) {
    const cleanPattern = pattern.replace(/\/$/, '')
    if (
      minimatch(relativePath, pattern, { dot: true }) ||
      minimatch(relativePath, `${cleanPattern}/**`, { dot: true }) ||
      relativePath.split('/').some((seg) => minimatch(seg, cleanPattern, { dot: true }))
    ) {
      return true
    }
  }
  return false
}
```

// src/parser/build-tree.ts
import { readdirSync, statSync, readFileSync } from 'fs'
import { join, relative } from 'path'
import type { GraphNode } from '../shared/types.js'
import { loadIgnorePatterns, shouldIgnore } from './ignore.js'

const LARGE_REPO_THRESHOLD = 5000

export function buildTree(rootPath: string): GraphNode[] {
  const patterns = loadIgnorePatterns(rootPath)
  const nodes: GraphNode[] = []
  walkDir(rootPath, rootPath, patterns, nodes, null)

  // Large repo safeguard: if > 5000 files, warn and strip file-level nodes
  const fileCount = nodes.filter((n) => n.type === 'file').length
  if (fileCount > LARGE_REPO_THRESHOLD) {
    console.warn(
      `Large repo detected (${fileCount} files). Starting with directory-level view only.`
    )
    // Keep only directory nodes; files will be lazy-loaded on expand
    return nodes
      .filter((n) => n.type === 'directory')
      .map((n) => ({
        ...n,
        children: n.children.filter(
          (cid) => nodes.find((nn) => nn.id === cid)?.type === 'directory'
        )
      }))
  }

  return nodes
}

function walkDir(
  dirPath: string,
  rootPath: string,
  patterns: ReturnType<typeof loadIgnorePatterns>,
  nodes: GraphNode[],
  parentId: string | null
): void {
  const entries = readdirSync(dirPath, { withFileTypes: true }).sort((a, b) =>
    a.name.localeCompare(b.name)
  )

  for (const entry of entries) {
    const fullPath = join(dirPath, entry.name)
    const relPath = relative(rootPath, fullPath)

    if (shouldIgnore(relPath, patterns)) continue

    if (entry.isDirectory()) {
      const dirNode: GraphNode = {
        id: relPath,
        label: entry.name,
        type: 'directory',
        children: [],
        parent: parentId,
        exports: [],
        lineCount: 0
      }
      nodes.push(dirNode)

      if (parentId !== null) {
        const parent = nodes.find((n) => n.id === parentId)
        if (parent) parent.children.push(dirNode.id)
      }

      walkDir(fullPath, rootPath, patterns, nodes, dirNode.id)
    } else if (entry.isFile()) {
      const content = readFileSync(fullPath, 'utf-8')
      const lineCount = content.split('\n').length

      const fileNode: GraphNode = {
        id: relPath,
        label: entry.name,
        type: 'file',
        children: [],
        parent: parentId,
        exports: [],
        lineCount
      }
      nodes.push(fileNode)

      if (parentId !== null) {
        const parent = nodes.find((n) => n.id === parentId)
        if (parent) parent.children.push(fileNode.id)
      }
    }
  }
}

// src/parser/parse-file.ts
import ts from 'typescript'
import { readFileSync } from 'fs'

const TS_EXTENSIONS = ['.ts', '.tsx', '.mts', '.cts']

export function parseFileExports(filePath: string): string[] {
  if (!TS_EXTENSIONS.some((ext) => filePath.endsWith(ext))) {
    return []
  }

  const content = readFileSync(filePath, 'utf-8')
  const sourceFile = ts.createSourceFile(filePath, content, ts.ScriptTarget.Latest, true)

  const exports: string[] = []

  ts.forEachChild(sourceFile, (node) => {
    // Direct exports: export class Foo, export function bar, export const X
    if (hasExportModifier(node)) {
      const name = getDeclarationName(node)
      if (name) exports.push(name)

      // export const A = ..., B = ...
      if (ts.isVariableStatement(node)) {
        for (const decl of node.declarationList.declarations) {
          if (ts.isIdentifier(decl.name)) {
            exports.push(decl.name.text)
          }
        }
      }
    }

    // Re-exports: export { Foo } from './bar'
    if (ts.isExportDeclaration(node) && node.exportClause) {
      if (ts.isNamedExports(node.exportClause)) {
        for (const spec of node.exportClause.elements) {
          exports.push((spec.name || spec.propertyName)!.text)
        }
      }
    }
  })

  return [...new Set(exports)]
}

function hasExportModifier(node: ts.Node): boolean {
  if (!ts.canHaveModifiers(node)) return false
  const mods = ts.getModifiers(node)
  return mods?.some((m) => m.kind === ts.SyntaxKind.ExportKeyword) ?? false
}

function getDeclarationName(node: ts.Node): string | undefined {
  if (
    ts.isFunctionDeclaration(node) ||
    ts.isClassDeclaration(node) ||
    ts.isInterfaceDeclaration(node) ||
    ts.isTypeAliasDeclaration(node) ||
    ts.isEnumDeclaration(node)
  ) {
    return node.name?.text
  }
  return undefined
}

// src/parser/resolve-edges.ts
import ts from 'typescript'
import { readdirSync } from 'fs'
import { join, relative, dirname } from 'path'
import type { GraphEdge } from '../shared/types.js'
import { loadIgnorePatterns, shouldIgnore } from './ignore.js'

export function resolveEdges(rootPath: string): GraphEdge[] {
  const patterns = loadIgnorePatterns(rootPath)
  const tsFiles = collectTsFiles(rootPath, rootPath, patterns)

  const configPath = ts.findConfigFile(rootPath, ts.sys.fileExists, 'tsconfig.json')
  const compilerOptions: ts.CompilerOptions = configPath
    ? readTsConfig(configPath)
    : {
        target: ts.ScriptTarget.ES2022,
        module: ts.ModuleKind.ESNext,
        moduleResolution: ts.ModuleResolutionKind.Bundler
      }

  const program = ts.createProgram(tsFiles, compilerOptions)
  const edgeMap = new Map<string, GraphEdge>()

  for (const sourceFile of program.getSourceFiles()) {
    const filePath = sourceFile.fileName
    if (!tsFiles.includes(filePath)) continue

    const sourceId = relative(rootPath, filePath)

    ts.forEachChild(sourceFile, (node) => {
      if (ts.isImportDeclaration(node) && ts.isStringLiteral(node.moduleSpecifier)) {
        const resolved = resolveModule(
          node.moduleSpecifier.text,
          filePath,
          compilerOptions,
          rootPath
        )
        if (!resolved) return

        const isTypeOnly = node.importClause?.isTypeOnly ?? false
        const type = isTypeOnly ? ('type-import' as const) : ('import' as const)
        const importCount = countImportedSymbols(node)

        addEdge(edgeMap, sourceId, resolved, type, importCount)
      }

      if (
        ts.isExportDeclaration(node) &&
        node.moduleSpecifier &&
        ts.isStringLiteral(node.moduleSpecifier)
      ) {
        const resolved = resolveModule(
          node.moduleSpecifier.text,
          filePath,
          compilerOptions,
          rootPath
        )
        if (!resolved) return

        const exportCount = countExportedSymbols(node)
        addEdge(edgeMap, sourceId, resolved, 're-export', exportCount)
      }
    })
  }

  return Array.from(edgeMap.values())
}

function addEdge(
  map: Map<string, GraphEdge>,
  source: string,
  target: string,
  type: GraphEdge['type'],
  symbolCount: number
): void {
  const key = `${source}→${target}→${type}`
  const existing = map.get(key)
  if (existing) {
    existing.weight = (existing.weight ?? 0) + symbolCount
  } else {
    map.set(key, { source, target, type, weight: symbolCount })
  }
}

function resolveModule(
  specifier: string,
  fromFile: string,
  options: ts.CompilerOptions,
  rootPath: string
): string | undefined {
  if (!specifier.startsWith('.')) return undefined

  const result = ts.resolveModuleName(specifier, fromFile, options, ts.sys)
  if (result.resolvedModule) {
    const resolvedPath = result.resolvedModule.resolvedFileName
    if (resolvedPath.includes('node_modules')) return undefined
    return relative(rootPath, resolvedPath)
  }
  return undefined
}

function countImportedSymbols(node: ts.ImportDeclaration): number {
  if (!node.importClause) return 1
  let count = 0
  if (node.importClause.name) count++
  if (node.importClause.namedBindings) {
    if (ts.isNamedImports(node.importClause.namedBindings)) {
      count += node.importClause.namedBindings.elements.length
    } else {
      count++
    }
  }
  return Math.max(count, 1)
}

function countExportedSymbols(node: ts.ExportDeclaration): number {
  if (node.exportClause && ts.isNamedExports(node.exportClause)) {
    return node.exportClause.elements.length
  }
  return 1
}

function readTsConfig(configPath: string): ts.CompilerOptions {
  const { config } = ts.readConfigFile(configPath, ts.sys.readFile)
  const parsed = ts.parseJsonConfigFileContent(config, ts.sys, dirname(configPath))
  return parsed.options
}

function collectTsFiles(
  dirPath: string,
  rootPath: string,
  patterns: ReturnType<typeof loadIgnorePatterns>
): string[] {
  const files: string[] = []
  const entries = readdirSync(dirPath, { withFileTypes: true })

  for (const entry of entries) {
    const fullPath = join(dirPath, entry.name)
    const relPath = relative(rootPath, fullPath)
    if (shouldIgnore(relPath, patterns)) continue

    if (entry.isDirectory()) {
      files.push(...collectTsFiles(fullPath, rootPath, patterns))
    } else if (entry.isFile() && /\.(ts|tsx|mts|cts)$/.test(entry.name)) {
      files.push(fullPath)
    }
  }
  return files
}

import { basename, join } from 'path'
import type { RepoGraph } from '../shared/types.js'
import { SCHEMA_VERSION } from '../shared/types.js'
import { buildTree } from './build-tree.js'
import { resolveEdges } from './resolve-edges.js'

export function parseRepo(rootPath: string): RepoGraph {
  const nodes = buildTree(rootPath)

  for (const node of nodes) {
    if (node.type === 'file') {
      const fullPath = join(rootPath, node.id)
      node.exports = parseFileExports(fullPath)
    }
  }

  const edges = resolveEdges(rootPath)

  const languages: string[] = []
  const hasTs = nodes.some((n) => n.type === 'file' && /\.(ts|tsx)$/.test(n.label))

  if (hasTs) languages.push('typescript')
  if (hasJs) languages.push('javascript')

  return {
    nodes,
    edges,
    metadata: {
      schemaVersion: SCHEMA_VERSION,
      name: basename(rootPath),
      rootPath,
      analyzedAt: new Date().toISOString(),
      languages,
      nodeCount: nodes.length,
      edgeCount: edges.length
    }
  }
}

// src/parser/cache.ts
import { existsSync, readFileSync, writeFileSync, mkdirSync, statSync, readdirSync } from 'fs'
import { join, relative } from 'path'
import type { RepoGraph } from '../shared/types.js'
import { SCHEMA_VERSION } from '../shared/types.js'
import { loadIgnorePatterns, shouldIgnore } from './ignore.js'

const CACHE_DIR = '.repomap'
const GRAPH_FILE = 'graph.json'

export function readCachedGraph(rootPath: string): RepoGraph | null {
  const cachePath = join(rootPath, CACHE_DIR, GRAPH_FILE)
  if (!existsSync(cachePath)) return null

  try {
    const raw = readFileSync(cachePath, 'utf-8')
    return JSON.parse(raw) as RepoGraph
  } catch {
    return null
  }
}

export function writeCachedGraph(rootPath: string, graph: RepoGraph): void {
  const cacheDir = join(rootPath, CACHE_DIR)
  mkdirSync(cacheDir, { recursive: true })
  writeFileSync(join(cacheDir, GRAPH_FILE), JSON.stringify(graph, null, 2))
}

export function isCacheStale(rootPath: string): boolean {
  const cached = readCachedGraph(rootPath)
  if (!cached) return true

  if (cached.metadata.schemaVersion !== SCHEMA_VERSION) return true

  const cachePath = join(rootPath, CACHE_DIR, GRAPH_FILE)
  const cacheMtime = statSync(cachePath).mtimeMs

  const patterns = loadIgnorePatterns(rootPath)
  const newestMtime = findNewestMtime(rootPath, rootPath, patterns)

  return newestMtime > cacheMtime
}

function findNewestMtime(
  dirPath: string,
  rootPath: string,
  patterns: ReturnType<typeof loadIgnorePatterns>
): number {
  let newest = 0
  const entries = readdirSync(dirPath, { withFileTypes: true })

  for (const entry of entries) {
    const fullPath = join(dirPath, entry.name)
    const relPath = relative(rootPath, fullPath)
    if (shouldIgnore(relPath, patterns)) continue

    if (entry.isDirectory()) {
      newest = Math.max(newest, findNewestMtime(fullPath, rootPath, patterns))
    } else if (entry.isFile()) {
      newest = Math.max(newest, statSync(fullPath).mtimeMs)
    }
  }

  return newest
}

// src/enrichment/cache.ts
import { createHash } from 'crypto'
import { existsSync, readFileSync, writeFileSync, mkdirSync } from 'fs'
import { join } from 'path'
import type { GraphNode, GraphEdge } from '../shared/types.js'

const CACHE_DIR = '.repomap'
const ENRICHMENT_FILE = 'enrichment.json'

export interface CachedEnrichment {
  hash: string
  summary: string
  domain: string
  tags: string[]
}

export interface CachedEdgeEnrichment {
  description: string
}

export interface EnrichmentCache {
  nodes: Record<string, CachedEnrichment>
  edges: Record<string, CachedEdgeEnrichment> // key: "source→target"
}

export function computeStructuralHash(node: GraphNode, edges: GraphEdge[]): string {
  const data = JSON.stringify({
    exports: node.exports.slice().sort(),
    edges: edges
      .filter((e) => e.source === node.id || e.target === node.id)
      .map((e) => `${e.source}→${e.target}:${e.type}`)
      .sort()
  })
  return createHash('sha256').update(data).digest('hex').slice(0, 16)
}

const EMPTY_CACHE: EnrichmentCache = { nodes: {}, edges: {} }

export function readEnrichmentCache(rootPath: string): EnrichmentCache {
  const cachePath = join(rootPath, CACHE_DIR, ENRICHMENT_FILE)
  if (!existsSync(cachePath)) return { ...EMPTY_CACHE }

  try {
    return JSON.parse(readFileSync(cachePath, 'utf-8'))
  } catch {
    return { ...EMPTY_CACHE }
  }
}

export function writeEnrichmentCache(rootPath: string, cache: EnrichmentCache): void {
  const cacheDir = join(rootPath, CACHE_DIR)
  mkdirSync(cacheDir, { recursive: true })
  writeFileSync(join(cacheDir, ENRICHMENT_FILE), JSON.stringify(cache, null, 2))
}

export function needsReenrichment(
  nodeId: string,
  currentHash: string,
  cache: EnrichmentCache
): boolean {
  const cached = cache.nodes[nodeId]
  if (!cached) return true
  return cached.hash !== currentHash
}

// src/enrichment/enrich-graph.ts
import type { RepoGraph, GraphNode, GraphEdge } from '../shared/types.js'
import type { EnrichmentAdapter, EnrichmentResult } from './ports.js'
import { buildEnrichmentPrompt } from './prompt-builder.js'
import { computeStructuralHash, type EnrichmentCache } from './cache.js'

const MAX_RETRIES = 1
const BATCH_SIZE = 20 // nodes per batch, grouped by directory proximity

export async function enrichGraph(
  graph: RepoGraph,
  adapter: EnrichmentAdapter,
  existingCache?: EnrichmentCache
): Promise<RepoGraph> {
  const cache = existingCache ?? { nodes: {}, edges: {} }
  const fileNodes = graph.nodes.filter((n) => n.type === 'file')

  // Filter to nodes that need enrichment (new or structurally changed)
  const needsEnrichment = fileNodes.filter((n) => {
    const hash = computeStructuralHash(n, graph.edges)
    const cached = cache.nodes[n.id]
    return !cached || cached.hash !== hash
  })

  if (needsEnrichment.length === 0) {
    console.log('All nodes already enriched, skipping LLM call.')
    return applyCache(graph, cache)
  }

  // Batch by directory proximity
  const batches = batchByDirectory(needsEnrichment, BATCH_SIZE)
  let successCount = 0
  let failCount = 0

  const allResults: EnrichmentResult = { nodes: [], edges: [] }

  for (const batch of batches) {
    const batchEdges = graph.edges.filter((e) =>
      batch.some((n) => n.id === e.source || n.id === e.target)
    )
    const prompt = buildEnrichmentPrompt(batch, batchEdges)

    let result: EnrichmentResult | null = null
    for (let attempt = 0; attempt <= MAX_RETRIES; attempt++) {
      try {
        result = await adapter.enrich(prompt)
        break
      } catch (err) {
        if (attempt === MAX_RETRIES) {
          console.warn(`Batch failed after ${MAX_RETRIES + 1} attempts:`, (err as Error).message)
          failCount += batch.length
        }
      }
    }

    if (result) {
      allResults.nodes.push(...result.nodes)
      allResults.edges.push(...result.edges)
      successCount += result.nodes.length
    }
  }

  console.log(`Enrichment: ${successCount} succeeded, ${failCount} failed`)

  // Apply results
  const enrichedNodes = graph.nodes.map((node) => {
    const enrichment = allResults.nodes.find((e) => e.nodeId === node.id)
    if (enrichment)
      return {
        ...node,
        summary: enrichment.summary,
        domain: enrichment.domain,
        tags: enrichment.tags
      }
    // Fall back to cache
    const cached = cache.nodes[node.id]
    if (cached)
      return { ...node, summary: cached.summary, domain: cached.domain, tags: cached.tags }
    return node
  })

  const enrichedEdges = graph.edges.map((edge) => {
    const enrichment = allResults.edges.find(
      (e) => e.source === edge.source && e.target === edge.target
    )
    if (enrichment) return { ...edge, description: enrichment.description }
    const cached = cache.edges[`${edge.source}→${edge.target}`]
    if (cached) return { ...edge, description: cached.description }
    return edge
  })

  return { nodes: enrichedNodes, edges: enrichedEdges, metadata: graph.metadata }
}

function applyCache(graph: RepoGraph, cache: EnrichmentCache): RepoGraph {
  const nodes = graph.nodes.map((n) => {
    const c = cache.nodes[n.id]
    return c ? { ...n, summary: c.summary, domain: c.domain, tags: c.tags } : n
  })
  const edges = graph.edges.map((e) => {
    const c = cache.edges[`${e.source}→${e.target}`]
    return c ? { ...e, description: c.description } : e
  })
  return { nodes, edges, metadata: graph.metadata }
}

function batchByDirectory(nodes: GraphNode[], batchSize: number): GraphNode[][] {
  // Sort by parent directory for proximity, then chunk
  const sorted = [...nodes].sort((a, b) => (a.parent ?? '').localeCompare(b.parent ?? ''))
  const batches: GraphNode[][] = []
  for (let i = 0; i < sorted.length; i += batchSize) {
    batches.push(sorted.slice(i, i + batchSize))
  }
  return batches
}

import { existsSync, readFileSync } from 'fs'
import { join } from 'path'
import { homedir, platform } from 'os'
import { execFileSync } from 'child_process'

interface AuthOptions {
  skipFileSystem?: boolean
}

export async function resolveAuthToken(options?: AuthOptions): Promise<string | null> {
  // 1. ANTHROPIC_AUTH_TOKEN env var
  const envToken = process.env.ANTHROPIC_AUTH_TOKEN
  if (envToken) return envToken

  if (options?.skipFileSystem) return null

  // 2. CLAUDE_TOKEN_FILE env var
  const tokenFilePath = process.env.CLAUDE_TOKEN_FILE
  if (tokenFilePath) {
    const token = readTokenFile(tokenFilePath)
    if (token) return token
  }

  // 3. ~/.claude/.credentials.json (cross-platform)
  const credentialsPath = join(homedir(), '.claude', '.credentials.json')
  if (existsSync(credentialsPath)) {
    try {
      const raw = JSON.parse(readFileSync(credentialsPath, 'utf-8'))
      const accessToken = raw?.claudeAiOauth?.accessToken
      if (accessToken) return accessToken
    } catch {
      // Malformed credentials file
    }
  }

  // 4. macOS Keychain (macOS only)
  if (platform() === 'darwin') {
    try {
      const raw = execFileSync(
        'security',
        ['find-generic-password', '-s', 'Claude Code-credentials', '-w'],
        { encoding: 'utf-8', timeout: 5000 }
      ).trim()
      const parsed = JSON.parse(raw)
      const accessToken = parsed?.claudeAiOauth?.accessToken
      if (accessToken) return accessToken
    } catch {
      // Keychain access failed
    }
  }

  return null
}

function readTokenFile(filePath: string): string | null {
  if (!existsSync(filePath)) return null
  try {
    const content = readFileSync(filePath, 'utf-8').trim()
    try {
      const parsed = JSON.parse(content)
      return parsed?.accessToken ?? parsed?.claudeAiOauth?.accessToken ?? content
    } catch {
      return content || null
    }
  } catch {
    return null
  }
}

        const m = msg as Record<string, unknown>
        if (m.type === 'assistant' && Array.isArray(m.content)) {
          for (const block of m.content) {
            if ((block as Record<string, unknown>).type === 'text') {

          }
        }

      // Parse JSON from response (may be wrapped in markdown code block)
      const jsonMatch =
        resultText.match(/```json\s*([\s\S]*?)```/) || resultText.match(/(\{[\s\S]*\})/)
      if (!jsonMatch) {
        throw new Error('LLM response did not contain valid JSON')
      }

      return JSON.parse(jsonMatch[1]) as EnrichmentResult
    }
  }

import type { EnrichmentCache } from '../enrichment/cache.js'

export function mergeGraphWithEnrichment(graph: RepoGraph, cache: EnrichmentCache): RepoGraph {
  const mergedNodes = graph.nodes.map((node) => {
    const enrichment = cache.nodes[node.id]
    if (!enrichment) return node
    return {
      ...node,
      summary: enrichment.summary,
      domain: enrichment.domain,
      tags: enrichment.tags
    }
  })

  const mergedEdges = graph.edges.map((edge) => {
    const enrichment = cache.edges[`${edge.source}→${edge.target}`]
    if (!enrichment) return edge
    return { ...edge, description: enrichment.description }
  })

  return {
    nodes: mergedNodes,
    edges: mergedEdges,
    metadata: graph.metadata
  }
}

import { resolve } from 'path'
import { parseRepo } from '../parser/parse-repo.js'
import { readCachedGraph, writeCachedGraph, isCacheStale } from '../parser/cache.js'
import {
  readEnrichmentCache,
  writeEnrichmentCache,
  computeStructuralHash,
  type EnrichmentCache
} from '../enrichment/cache.js'

import type { RepoGraph } from '../shared/types.js'

export interface CliOptions {
  targetPath: string
  port: number
  noEnrich: boolean
  json: boolean
  force: boolean
}

export function parseArgs(args: string[]): CliOptions {
  const opts: CliOptions = {
    targetPath: '.',
    port: 3000,
    noEnrich: false,
    json: false,
    force: false
  }

  let i = 0
  while (i < args.length) {
    const arg = args[i]
    if (arg === '--port' && args[i + 1]) {
      opts.port = parseInt(args[i + 1], 10)
      i += 2
    } else if (arg === '--no-enrich') {
      opts.noEnrich = true
      i++
    } else if (arg === '--json') {
      opts.json = true
      i++
    } else if (arg === '--force') {
      opts.force = true
      i++
    } else if (!arg.startsWith('--')) {
      opts.targetPath = arg
      i++
    } else {
      i++
    }
  }

  return opts
}

export async function run(
  opts: CliOptions,
  deps?: { enrichmentAdapter?: EnrichmentAdapter }
): Promise<RepoGraph> {
  const rootPath = resolve(opts.targetPath)

  // Step 1: Parse (with caching)
  let graph: RepoGraph
  if (!opts.force && !isCacheStale(rootPath)) {
    graph = readCachedGraph(rootPath)!
    console.log('Using cached structural graph.')
  } else {
    console.log('Parsing repository...')
    graph = parseRepo(rootPath)
    writeCachedGraph(rootPath, graph)
    console.log(`Parsed ${graph.metadata.nodeCount} nodes, ${graph.metadata.edgeCount} edges.`)
  }

  // Step 2: Enrich (if not skipped)
  const enrichmentCache = readEnrichmentCache(rootPath)

  if (!opts.noEnrich && deps?.enrichmentAdapter) {
    console.log('Enriching graph with LLM annotations...')
    const enriched = await enrichGraph(graph, deps.enrichmentAdapter, enrichmentCache)

    // Update cache with new enrichment results
    const newCache: EnrichmentCache = {
      nodes: { ...enrichmentCache.nodes },
      edges: { ...enrichmentCache.edges }
    }
    for (const node of enriched.nodes) {
      if (node.summary) {
        const hash = computeStructuralHash(node, enriched.edges)
        newCache.nodes[node.id] = {
          hash,
          summary: node.summary,
          domain: node.domain ?? '',
          tags: node.tags ?? []
        }
      }
    }
    for (const edge of enriched.edges) {
      if (edge.description) {
        newCache.edges[`${edge.source}→${edge.target}`] = {
          description: edge.description
        }
      }
    }
    writeEnrichmentCache(rootPath, newCache)

    return enriched
  }

  // Step 3: Merge cached enrichment
  return mergeGraphWithEnrichment(graph, enrichmentCache)
}

// src/server/server.ts
import express from 'express'
import { createServer as createHttpServer, type Server } from 'http'
import { WebSocketServer, type WebSocket } from 'ws'
import { createGraphHandler } from './graph-api.js'
import type { RepoGraph, GraphDelta } from '../shared/types.js'

interface ServerOptions {
  getGraph: () => RepoGraph
  port: number
  staticDir?: string
}

interface ServerHandle {
  server: Server
  port: number
  wss: WebSocketServer
  broadcastDelta: (delta: GraphDelta) => void
  close: () => void
}

export async function createServer(opts: ServerOptions): Promise<ServerHandle> {
  const app = express()
  app.use(express.json())

  app.get('/api/graph', createGraphHandler(opts.getGraph))

  if (opts.staticDir) {
    app.use(express.static(opts.staticDir))
  }

  const server = createHttpServer(app)
  const wss = new WebSocketServer({ server })

  const clients = new Set<WebSocket>()
  wss.on('connection', (ws) => {
    clients.add(ws)
    ws.on('close', () => clients.delete(ws))
  })

  function broadcastDelta(delta: GraphDelta) {
    const msg = JSON.stringify(delta)
    for (const client of clients) {
      if (client.readyState === client.OPEN) {
        client.send(msg)
      }
    }
  }

  return new Promise((resolve, reject) => {
    server.on('error', (err: NodeJS.ErrnoException) => {
      if (err.code === 'EADDRINUSE') {
        reject(new Error(`Port ${opts.port} is in use. Try --port ${opts.port + 1}`))
      } else {
        reject(err)
      }
    })

    server.listen(opts.port, () => {
      const addr = server.address()
      const actualPort = typeof addr === 'object' && addr ? addr.port : opts.port

      resolve({
        server,
        port: actualPort,
        wss,
        broadcastDelta,
        close: () => {
          for (const client of clients) client.close()
          wss.close()
          server.close()
        }
      })
    })
  })
}

// src/server/watcher.ts
import { watch } from 'chokidar'
import { relative } from 'path'
import type { RepoGraph, GraphDelta } from '../shared/types.js'
import { parseRepo } from '../parser/parse-repo.js'
import { loadIgnorePatterns, shouldIgnore } from '../parser/ignore.js'

export function computeDelta(before: RepoGraph, after: RepoGraph): GraphDelta {
  const beforeNodeMap = new Map(before.nodes.map((n) => [n.id, n]))
  const afterNodeMap = new Map(after.nodes.map((n) => [n.id, n]))

  const nodesAdded = after.nodes.filter((n) => !beforeNodeMap.has(n.id))
  const nodesRemoved = before.nodes.filter((n) => !afterNodeMap.has(n.id)).map((n) => n.id)
  const nodesUpdated = after.nodes.filter((n) => {
    const prev = beforeNodeMap.get(n.id)
    if (!prev) return false
    return JSON.stringify(prev) !== JSON.stringify(n)
  })

  const edgeKey = (e: { source: string; target: string }) => `${e.source}→${e.target}`
  const beforeEdgeSet = new Set(before.edges.map(edgeKey))
  const afterEdgeSet = new Set(after.edges.map(edgeKey))

  const edgesAdded = after.edges.filter((e) => !beforeEdgeSet.has(edgeKey(e)))
  const edgesRemoved = before.edges
    .filter((e) => !afterEdgeSet.has(edgeKey(e)))
    .map((e) => ({ source: e.source, target: e.target }))
  const edgesUpdated = after.edges.filter((e) => {
    if (!beforeEdgeSet.has(edgeKey(e))) return false
    const prev = before.edges.find((be) => edgeKey(be) === edgeKey(e))
    return prev && JSON.stringify(prev) !== JSON.stringify(e)
  })

  return { nodesAdded, nodesRemoved, nodesUpdated, edgesAdded, edgesRemoved, edgesUpdated }
}

export interface WatcherHandle {
  close: () => Promise<void>
}

export function startWatcher(
  rootPath: string,
  currentGraph: { get: () => RepoGraph; set: (g: RepoGraph) => void },
  onDelta: (delta: GraphDelta) => void
): WatcherHandle {
  const patterns = loadIgnorePatterns(rootPath)
  let debounceTimer: ReturnType<typeof setTimeout> | null = null

  const watcher = watch(rootPath, {
    ignored: (path: string) => {
      const rel = relative(rootPath, path)
      return shouldIgnore(rel, patterns)
    },
    persistent: true,
    ignoreInitial: true
  })

  const handleChange = () => {
    if (debounceTimer) clearTimeout(debounceTimer)
    debounceTimer = setTimeout(() => {
      try {
        const before = currentGraph.get()
        const after = parseRepo(rootPath)
        const delta = computeDelta(before, after)

        const hasChanges =
          delta.nodesAdded.length > 0 ||
          delta.nodesRemoved.length > 0 ||
          delta.nodesUpdated.length > 0 ||
          delta.edgesAdded.length > 0 ||
          delta.edgesRemoved.length > 0

        if (hasChanges) {
          currentGraph.set(after)
          onDelta(delta)
        }
      } catch (err) {
        console.error('Watcher re-parse failed:', (err as Error).message)
      }
    }, 500)
  }

  watcher.on('change', handleChange)
  watcher.on('add', handleChange)
  watcher.on('unlink', handleChange)

  return {
    close: async () => {
      if (debounceTimer) clearTimeout(debounceTimer)
      await watcher.close()
    }
  }
}
```
