# IDE View Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the Terminal view (⌘3) with a full IDE view combining file explorer sidebar, Monaco code editor, and integrated terminal panel.

**Architecture:** Three-zone layout using `react-resizable-panels` (existing dependency). File explorer sidebar on the left, tabbed Monaco editor in the center, and the existing terminal implementation in a collapsible bottom panel. All file I/O goes through new IPC channels with path validation. State managed by a new `ideStore` Zustand store with debounced persistence to SQLite.

**Tech Stack:** Monaco Editor via `@monaco-editor/react`, existing xterm.js/node-pty terminal, React + TypeScript + Zustand, `react-resizable-panels` for layout. Dependencies approved during design review.

**Spec:** `docs/superpowers/specs/2026-03-24-ide-view-design.md`

```typescript
import { ipcMain, shell, BrowserWindow } from 'electron'
import * as fs from 'node:fs/promises'
import * as fsSync from 'node:fs'
import * as path from 'node:path'
import { safeHandle } from '../ipc-utils'

const MAX_FILE_SIZE = 5 * 1024 * 1024 // 5MB
const BINARY_CHECK_SIZE = 8192 // 8KB
const DEBOUNCE_MS = 500

let currentWatcher: fsSync.FSWatcher | null = null
let ideRootPath: string | null = null

/** Validates that targetPath is within allowedRoot. Returns resolved path. */
export function validateIdePath(targetPath: string, allowedRoot: string): string {
  const resolved = path.resolve(targetPath)
  const root = path.resolve(allowedRoot)
  if (!resolved.startsWith(root + path.sep) && resolved !== root) {
    throw new Error(`Path "${targetPath}" is outside allowed root`)
  }
  return resolved
}

/** Reads directory contents, sorted folders-first then alphabetical. */
export async function readDir(
  dirPath: string
): Promise<{ name: string; type: 'file' | 'directory'; size: number }[]> {
  const entries = await fs.readdir(dirPath, { withFileTypes: true })
  const results = await Promise.all(
    entries.map(async (entry) => {
      const fullPath = path.join(dirPath, entry.name)
      let size = 0
      try {
        const stat = await fs.stat(fullPath)
        size = stat.size
      } catch {
        // Ignore stat errors (broken symlinks, permission issues)
      }
      return {
        name: entry.name,
        type: (entry.isDirectory() ? 'directory' : 'file') as 'file' | 'directory',
        size
      }
    })
  )
  return results.sort((a, b) => {
    if (a.type !== b.type) return a.type === 'directory' ? -1 : 1
    return a.name.localeCompare(b.name)
  })
}

/** Reads file content as UTF-8 with size and binary guards. */
export async function readFileContent(filePath: string): Promise<string> {
  const stat = await fs.stat(filePath)
  if (stat.size > MAX_FILE_SIZE) {
    throw new Error(`File too large (${(stat.size / 1024 / 1024).toFixed(1)}MB). Maximum is 5MB.`)
  }

  const fd = await fs.open(filePath, 'r')
  try {
    const checkBuf = Buffer.alloc(Math.min(BINARY_CHECK_SIZE, stat.size))
    await fd.read(checkBuf, 0, checkBuf.length, 0)
    if (checkBuf.includes(0)) {
      throw new Error('Cannot open binary file')
    }
  } finally {
    await fd.close()
  }

  return fs.readFile(filePath, 'utf-8')
}

/** Writes file atomically (write to temp, then rename). */
export async function writeFileContent(filePath: string, content: string): Promise<void> {
  const dir = path.dirname(filePath)
  const tmpPath = path.join(dir, `.${path.basename(filePath)}.tmp.${Date.now()}`)
  await fs.writeFile(tmpPath, content, 'utf-8')
  await fs.rename(tmpPath, filePath)
}

function requireRoot(): string {
  if (!ideRootPath) throw new Error('No IDE folder is open')
  return ideRootPath
}

function startWatcher(rootPath: string): void {
  stopWatcher()

  let debounceTimer: ReturnType<typeof setTimeout> | null = null
  const changedPaths = new Set<string>()

  currentWatcher = fsSync.watch(rootPath, { recursive: true }, (_event, filename) => {
    if (filename) {
      changedPaths.add(path.dirname(path.join(rootPath, filename)))
    }
    if (debounceTimer) clearTimeout(debounceTimer)
    debounceTimer = setTimeout(() => {
      const paths = [...changedPaths]
      changedPaths.clear()
      const windows = BrowserWindow.getAllWindows()
      for (const win of windows) {
        for (const p of paths) {
          win.webContents.send('fs:dirChanged', p)
        }
      }
    }, DEBOUNCE_MS)
  })
}

function stopWatcher(): void {
  if (currentWatcher) {
    currentWatcher.close()
    currentWatcher = null
  }
}

export function registerIdeFsHandlers(): void {
  safeHandle('fs:readDir', async (_e, dirPath: string) => {
    const root = requireRoot()
    validateIdePath(dirPath, root)
    return readDir(dirPath)
  })

  safeHandle('fs:readFile', async (_e, filePath: string) => {
    const root = requireRoot()
    validateIdePath(filePath, root)
    return readFileContent(filePath)
  })

  safeHandle('fs:writeFile', async (_e, filePath: string, content: string) => {
    const root = requireRoot()
    validateIdePath(filePath, root)
    await writeFileContent(filePath, content)
  })

  safeHandle('fs:watchDir', async (_e, dirPath: string) => {
    ideRootPath = dirPath
    startWatcher(dirPath)
  })

  safeHandle('fs:unwatchDir', async () => {
    stopWatcher()
    ideRootPath = null
  })

  safeHandle('fs:createFile', async (_e, filePath: string) => {
    const root = requireRoot()
    validateIdePath(filePath, root)
    await fs.writeFile(filePath, '', 'utf-8')
  })

  safeHandle('fs:createDir', async (_e, dirPath: string) => {
    const root = requireRoot()
    validateIdePath(dirPath, root)
    await fs.mkdir(dirPath, { recursive: true })
  })

  safeHandle('fs:rename', async (_e, oldPath: string, newPath: string) => {
    const root = requireRoot()
    validateIdePath(oldPath, root)
    validateIdePath(newPath, root)
    await fs.rename(oldPath, newPath)
  })

  safeHandle('fs:delete', async (_e, targetPath: string) => {
    const root = requireRoot()
    validateIdePath(targetPath, root)
    await shell.trashItem(targetPath)
  })

  safeHandle('fs:stat', async (_e, targetPath: string) => {
    const root = requireRoot()
    validateIdePath(targetPath, root)
    const stat = await fs.stat(targetPath)
    return {
      size: stat.size,
      mtime: stat.mtimeMs,
      isDirectory: stat.isDirectory()
    }
  })
}
```

```typescript
  language: string
  isDirty: boolean
}

interface IDEState {
  rootPath: string | null
  expandedDirs: Record<string, boolean>
  openTabs: EditorTab[]
  activeTabId: string | null
  focusedPanel: 'editor' | 'terminal'
  sidebarCollapsed: boolean
  terminalCollapsed: boolean
  sidebarWidth: number
  terminalHeight: number
  recentFolders: string[]

  setRootPath: (rootPath: string | null) => void
  toggleDir: (dirPath: string) => void
  openTab: (filePath: string) => void
  closeTab: (tabId: string) => void
  setActiveTab: (tabId: string) => void
  setDirty: (tabId: string, dirty: boolean) => void
  reorderTabs: (fromIndex: number, toIndex: number) => void
  toggleSidebar: () => void
  toggleTerminal: () => void
  setFocusedPanel: (panel: 'editor' | 'terminal') => void
  setSidebarWidth: (width: number) => void
  setTerminalHeight: (height: number) => void
}

const MAX_RECENT_FOLDERS = 5

function detectLanguage(filePath: string): string {
  const ext = filePath.split('.').pop()?.toLowerCase() ?? ''
  const langMap: Record<string, string> = {
    ts: 'typescript', tsx: 'typescript',
    js: 'javascript', jsx: 'javascript',
    json: 'json', md: 'markdown',
    css: 'css', scss: 'scss', html: 'html',
    py: 'python', rs: 'rust', go: 'go',
    yaml: 'yaml', yml: 'yaml', toml: 'toml',
    sh: 'shell', bash: 'shell', zsh: 'shell',
    sql: 'sql', xml: 'xml', svg: 'xml'
  }
  return langMap[ext] ?? 'plaintext'
}

export const useIDEStore = create<IDEState>((set, get) => ({
  rootPath: null,
  expandedDirs: {},
  openTabs: [],
  activeTabId: null,
  focusedPanel: 'editor',
  sidebarCollapsed: false,
  terminalCollapsed: false,
  sidebarWidth: 240,
  terminalHeight: 35,
  recentFolders: [],

  setRootPath: (rootPath) => {
    const recent = get().recentFolders
    let updated = recent
    if (rootPath) {
      updated = [rootPath, ...recent.filter((f) => f !== rootPath)].slice(0, MAX_RECENT_FOLDERS)
    }
    set({ rootPath, expandedDirs: {}, recentFolders: updated })
  },

  toggleDir: (dirPath) =>
    set((s) => ({
      expandedDirs: { ...s.expandedDirs, [dirPath]: !s.expandedDirs[dirPath] }
    })),

  openTab: (filePath) => {
    const state = get()
    const existing = state.openTabs.find((t) => t.filePath === filePath)
    if (existing) {
      set({ activeTabId: existing.id })
      return
    }
    const tab: EditorTab = {
      id: `tab-${Date.now()}-${Math.random().toString(36).slice(2, 7)}`,
      filePath,
      displayName: filePath.split('/').pop() ?? filePath,
      language: detectLanguage(filePath),
      isDirty: false
    }
    set({ openTabs: [...state.openTabs, tab], activeTabId: tab.id })
  },

  closeTab: (tabId) => {
    const state = get()
    const idx = state.openTabs.findIndex((t) => t.id === tabId)
    if (idx === -1) return
    const newTabs = state.openTabs.filter((t) => t.id !== tabId)
    let newActiveId = state.activeTabId
    if (state.activeTabId === tabId) {
      if (newTabs.length === 0) {
        newActiveId = null
      } else if (idx >= newTabs.length) {
        newActiveId = newTabs[newTabs.length - 1].id
      } else {
        newActiveId = newTabs[idx].id
      }
    }
    set({ openTabs: newTabs, activeTabId: newActiveId })
  },

  setActiveTab: (tabId) => set({ activeTabId: tabId }),

  setDirty: (tabId, dirty) =>
    set((s) => ({
      openTabs: s.openTabs.map((t) => (t.id === tabId ? { ...t, isDirty: dirty } : t))
    })),

  reorderTabs: (fromIndex, toIndex) =>
    set((s) => {
      const tabs = [...s.openTabs]
      const [moved] = tabs.splice(fromIndex, 1)
      tabs.splice(toIndex, 0, moved)
      return { openTabs: tabs }
    }),

  toggleSidebar: () => set((s) => ({ sidebarCollapsed: !s.sidebarCollapsed })),
  toggleTerminal: () => set((s) => ({ terminalCollapsed: !s.terminalCollapsed })),
  setFocusedPanel: (panel) => set({ focusedPanel: panel }),
  setSidebarWidth: (width) => set({ sidebarWidth: width }),
  setTerminalHeight: (height) => set({ terminalHeight: height })
}))
```

Note: No `import * as path from 'path'` — renderer is sandboxed. Use string splitting (`filePath.split('/').pop()`) instead.

```tsx
import { useRef, useEffect, useCallback } from 'react'
import MonacoEditor, { loader } from '@monaco-editor/react'
import type { editor } from 'monaco-editor'
import { useThemeStore } from '../../stores/theme'
import { getMonacoTheme, getLightMonacoTheme } from '../../lib/monaco-theme'

interface EditorPaneProps {
  filePath: string | null
  content: string | null
  language: string
  onContentChange?: (content: string) => void
  onSave?: () => void
}

export default function EditorPane({
  filePath,
  content,
  language,
  onContentChange,
  onSave
}: EditorPaneProps): JSX.Element {
  const editorRef = useRef<editor.IStandaloneCodeEditor | null>(null)
  const monacoRef = useRef<typeof import('monaco-editor') | null>(null)
  const theme = useThemeStore((s) => s.theme)

  const handleMount = useCallback(
    (editorInstance: editor.IStandaloneCodeEditor, monaco: typeof import('monaco-editor')) => {
      editorRef.current = editorInstance
      monacoRef.current = monaco

      // Register BDE themes
      monaco.editor.defineTheme('bde-dark', getMonacoTheme())
      monaco.editor.defineTheme('bde-light', getLightMonacoTheme())
      monaco.editor.setTheme(theme === 'dark' ? 'bde-dark' : 'bde-light')

      // ⌘S save keybinding
      editorInstance.addCommand(monaco.KeyMod.CtrlCmd | monaco.KeyCode.KeyS, () => {
        onSave?.()
      })

      editorInstance.focus()
    },
    [theme, onSave]
  )

  // Sync theme changes
  useEffect(() => {
    if (monacoRef.current) {
      monacoRef.current.editor.defineTheme('bde-dark', getMonacoTheme())
      monacoRef.current.editor.defineTheme('bde-light', getLightMonacoTheme())
      monacoRef.current.editor.setTheme(theme === 'dark' ? 'bde-dark' : 'bde-light')
    }
  }, [theme])

  if (!filePath || content === null) {
    return (
      <div className="ide-editor-empty">
        <p>Open a file from the sidebar to start editing</p>
      </div>
    )
  }

  return (
    <MonacoEditor
      height="100%"
      language={language}
      value={content}
      theme={theme === 'dark' ? 'bde-dark' : 'bde-light'}
      onChange={(value) => onContentChange?.(value ?? '')}
      onMount={handleMount}
      options={{
        fontSize: 13,
        fontFamily: "'JetBrains Mono', 'Fira Code', 'Cascadia Code', monospace",
        minimap: { enabled: true },
        scrollBeyondLastLine: false,
        automaticLayout: true,
        tabSize: 2,
        wordWrap: 'off',
        renderWhitespace: 'selection',
        bracketPairColorization: { enabled: true },
        padding: { top: 8 }
      }}
    />
  )
}
```

```tsx
import { useIDEStore } from '../../stores/ide'
import { X } from 'lucide-react'

interface EditorTabBarProps {
  onCloseTab?: (tabId: string, isDirty: boolean) => void
}

export default function EditorTabBar({ onCloseTab }: EditorTabBarProps): JSX.Element {
  const openTabs = useIDEStore((s) => s.openTabs)
  const activeTabId = useIDEStore((s) => s.activeTabId)
  const setActiveTab = useIDEStore((s) => s.setActiveTab)
  const closeTab = useIDEStore((s) => s.closeTab)

  const handleClose = (e: React.MouseEvent, tabId: string, isDirty: boolean): void => {
    e.stopPropagation()
    if (onCloseTab) {
      onCloseTab(tabId, isDirty)
    } else {
      closeTab(tabId)
    }
  }

  const handleMouseDown = (e: React.MouseEvent, tabId: string): void => {
    if (e.button === 1) {
      e.preventDefault()
      const tab = openTabs.find((t) => t.id === tabId)
      if (tab) handleClose(e, tabId, tab.isDirty)
    }
  }

  return (
    <div className="ide-editor-tab-bar" role="tablist" aria-label="Open files">
      {openTabs.map((tab) => (
        <div
          key={tab.id}
          className={`ide-editor-tab ${tab.id === activeTabId ? 'ide-editor-tab--active' : ''}`}
          role="tab"
          aria-selected={tab.id === activeTabId}
          title={tab.filePath}
          onClick={() => setActiveTab(tab.id)}
          onMouseDown={(e) => handleMouseDown(e, tab.id)}
        >
          <span className="ide-editor-tab__name">
            {tab.isDirty && <span className="ide-editor-tab__dirty">●</span>}
            {tab.displayName}
          </span>
          <button
            className="ide-editor-tab__close"
            aria-label="Close tab"
            onClick={(e) => handleClose(e, tab.id, tab.isDirty)}
          >
            <X size={14} />
          </button>
        </div>
      ))}
    </div>
  )
}
```

```tsx
function getFileIcon(name: string): JSX.Element {
  const ext = name.split('.').pop()?.toLowerCase() ?? ''
  switch (ext) {
    case 'ts': case 'tsx': case 'js': case 'jsx': case 'py': case 'rs': case 'go':
      return <FileCode size={16} />
    case 'json':
      return <FileJson size={16} />
    case 'md': case 'txt':
      return <FileText size={16} />
    default:
      return <File size={16} />
  }
}

export default function FileTreeNode({
  entry, parentPath, depth, onFileClick, onContextMenu
}: FileTreeNodeProps): JSX.Element {
  const fullPath = `${parentPath}/${entry.name}`
  const isExpanded = useIDEStore((s) => s.expandedDirs[fullPath] ?? false)
  const activeTabId = useIDEStore((s) => s.activeTabId)
  const openTabs = useIDEStore((s) => s.openTabs)
  const toggleDir = useIDEStore((s) => s.toggleDir)

  const [children, setChildren] = useState<DirEntry[] | null>(null)
  const [loading, setLoading] = useState(false)

    if (entry.type === 'directory') {
      toggleDir(fullPath)

          const entries = await window.api.readDir(fullPath)
          setChildren(entries.filter((e: DirEntry) => !(e.type === 'directory' && HIDDEN_DIRS.has(e.name))))
        } catch {
          setChildren([])
        }
        setLoading(false)
      }
    } else {
      onFileClick(fullPath)
    }
  }, [entry.type, fullPath, isExpanded, children, toggleDir, onFileClick])

  return (
```

```tsx
  onContextMenu: (e: React.MouseEvent, filePath: string, isDir: boolean) => void

export default function FileTree({ dirPath, onFileClick, onContextMenu }: FileTreeProps): JSX.Element {
  const [entries, setEntries] = useState<DirEntry[]>([])
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    let cancelled = false
    const load = async (): Promise<void> => {
      try {
        const result = await window.api.readDir(dirPath)
        if (!cancelled) {
          setEntries(result.filter((e: DirEntry) => !(e.type === 'directory' && HIDDEN_DIRS.has(e.name))))
          setError(null)
        }
      } catch (err) {
        if (!cancelled) {
          setError(err instanceof Error ? err.message : 'Failed to read directory')
        }
      }
    }
    load()

    const unsub = window.api.onDirChanged?.((changedPath: string) => {
      if (changedPath === dirPath || changedPath.startsWith(dirPath)) {
        load()
      }
    })

    return () => {
      cancelled = true
      unsub?.()
    }
  }, [dirPath])

  if (error) return <div className="ide-file-tree__error">{error}</div>

  return (
    <div className="ide-file-tree" role="tree" aria-label="File explorer">
      {entries.map((entry) => (
        <FileTreeNode
          key={entry.name}
          entry={entry}
          parentPath={dirPath}
          depth={0}
          onFileClick={onFileClick}
          onContextMenu={onContextMenu}
        />
      ))}
    </div>
  )
}
```

```tsx
            e.preventDefault(); e.stopPropagation()
            useTerminalStore.getState().zoomIn()
            return
          }
          if (e.key === '-') {
            e.preventDefault(); e.stopPropagation()
            useTerminalStore.getState().zoomOut()
            return
          }
          if (e.key === '0') {
            e.preventDefault(); e.stopPropagation()
            useTerminalStore.getState().resetZoom()
            return
          }
        }
      }

      // Ctrl+L — Clear terminal (when terminal focused)
      if (e.ctrlKey && e.key === 'l' && !e.metaKey && focusedPanel === 'terminal') {
        e.preventDefault(); e.stopPropagation()
        const termActiveTabId = useTerminalStore.getState().activeTabId
        const termActiveTab = useTerminalStore.getState().tabs.find((t) => t.id === termActiveTabId)
        if (termActiveTabId && termActiveTab?.kind !== 'agent') {
          clearTerminal(termActiveTabId)
        }
        return
      }
    }

    window.addEventListener('keydown', handler, true)
    return () => window.removeEventListener('keydown', handler, true)
  }, [activeView, focusedPanel, activeTab, toggleSidebar, toggleTerminal, handleOpenFolder, handleSave, handleCloseTab])

  // No folder open — show empty state
  if (!rootPath) {
    return <IDEEmptyState onOpenFolder={handleOpenFolder} recentFolders={recentFolders} onOpenRecent={handleOpenRecent} />
  }

  return (
    <div className="ide-view">
      <Group orientation="horizontal">
        {!sidebarCollapsed && (
          <>
            <Panel defaultSize={20} minSize={10} maxSize={40}>
              <FileSidebar />
            </Panel>
            <Separator className="ide-separator" />
          </>
        )}
        <Panel>
          <Group orientation="vertical">
            <Panel defaultSize={terminalCollapsed ? 100 : 65} onClick={() => setFocusedPanel('editor')}>
              <div className="ide-editor-area">
                <EditorTabBar onCloseTab={handleCloseTab} />
                <EditorPane
                  filePath={activeTab?.filePath ?? null}
                  content={activeTab ? (fileContents[activeTab.filePath] ?? null) : null}
                  language={activeTab?.language ?? 'plaintext'}
                  onContentChange={handleContentChange}
                  onSave={handleSave}
                />
              </div>
            </Panel>
            {!terminalCollapsed && (
              <>
                <Separator className="ide-separator" />
                <Panel defaultSize={35} minSize={10} maxSize={60}>
                  <TerminalPanel />
                </Panel>
              </>
            )}
          </Group>
        </Panel>
      </Group>
      <ConfirmModal {...confirmProps} />
    </div>
  )
}
```

At the bottom of `src/renderer/src/stores/ide.ts`, after the store definition, add a subscriber that persists state with 2-second debounce. Use `window.api.settings.setJson()` (the actual API — see preload line 36):

```typescript
let persistTimer: ReturnType<typeof setTimeout> | null = null

useIDEStore.subscribe((state) => {
  if (persistTimer) clearTimeout(persistTimer)
  persistTimer = setTimeout(() => {
    const toSave = {
      rootPath: state.rootPath,
      openTabs: state.openTabs.map((t) => ({ filePath: t.filePath })),
      activeFilePath: state.openTabs.find((t) => t.id === state.activeTabId)?.filePath ?? null,
      sidebarCollapsed: state.sidebarCollapsed,
      terminalCollapsed: state.terminalCollapsed,
      recentFolders: state.recentFolders
    }
    window.api.settings.setJson('ide.state', toSave)
  }, 2000)
})
```

Note: We persist `activeFilePath` (not `activeTabId`) because tab IDs are randomly generated on each session. On restore, we match by file path.

- [ ] **Step 2: Add state restoration in IDEView.tsx**

Add a `useEffect` at the top of the IDEView component:

```typescript
useEffect(() => {
  const restore = async (): Promise<void> => {
    try {
      const saved = await window.api.settings.getJson('ide.state')
      if (!saved || typeof saved !== 'object') return
      const state = saved as {
        rootPath?: string
        openTabs?: { filePath: string }[]
        activeFilePath?: string
        sidebarCollapsed?: boolean
        terminalCollapsed?: boolean
        recentFolders?: string[]
      }
      useIDEStore.setState({
        rootPath: state.rootPath ?? null,
        sidebarCollapsed: state.sidebarCollapsed ?? false,
        terminalCollapsed: state.terminalCollapsed ?? false,
        recentFolders: state.recentFolders ?? []
      })
      if (state.openTabs) {
        for (const tab of state.openTabs) {
          useIDEStore.getState().openTab(tab.filePath)
        }
        // Restore active tab by matching file path
        if (state.activeFilePath) {
          const match = useIDEStore.getState().openTabs.find((t) => t.filePath === state.activeFilePath)
          if (match) useIDEStore.getState().setActiveTab(match.id)
        }
      }
      if (state.rootPath) {
        await window.api.watchDir(state.rootPath)
      }
    } catch (err) {
      console.error('Failed to restore IDE state:', err)
    }
  }
  restore()
}, [])
```
